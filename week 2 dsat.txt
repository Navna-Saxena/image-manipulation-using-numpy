## flipping image ##

import matplotlib.pyplot as plt
import numpy as np

im = np.flipud(plt.imread('PSIT kanpur.png'))
plt.subplot(2, 1, 1)
plt.imshow(im)
plt.subplot(2, 1, 2)
plt.imshow(np.fliplr(im))
plt.show()
